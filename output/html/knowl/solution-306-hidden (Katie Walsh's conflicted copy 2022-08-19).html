\begin{equation*} \displaystyle{\int_{a}^{b} f(x)dx} =\displaystyle{\int_{a}^{c} f(x)dx} +\displaystyle{\int_{c}^{b} f(x)dx} . \end{equation*}

Thus,

\begin{align*} \displaystyle{\int_{0}^{8} f(x)dx} \amp = \displaystyle{\int_{0}^{5} f(x)dx} +\displaystyle{\int_{5}^{8} f(x)dx} \\ 10\amp = 5+\displaystyle{\int_{5}^{8} f(x)dx} \\ 5\amp = \displaystyle{\int_{5}^{8} f(x)dx} . \end{align*}
