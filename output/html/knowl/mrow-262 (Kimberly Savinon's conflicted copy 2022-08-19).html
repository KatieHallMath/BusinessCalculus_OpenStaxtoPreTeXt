\begin{align*} \lim_{x\to\infty}\frac{3x-2}{\sqrt{4x^2+5}}\amp = \lim_{x\to\infty}\frac{(1/|x|)}{(3x-2)}{(1/|x|)\sqrt{4x^2+5}}\\ \amp=\lim_{x\to\infty}\frac{(1/x)}{(3x-2)}{\sqrt{(1/x^2)}{(4x^2+5)}}\\ \amp= \lim_{x\to\infty}\frac{3-2/x}{\sqrt{4+5/x^2}}=\frac{3}{\sqrt{4}}=\frac{3}{2}\\ \lim_{x\to-\infty}\frac{3x-2}{\sqrt{4x^2+5}}\amp = \lim_{x\to-\infty}\frac{(1/|x|)}{(3x-2)}{(1/|x|)\sqrt{4x^2+5}}\\ \amp= \lim_{x\to-\infty}\frac{(-1/x)}{(3x-2)}{\sqrt{(1/x^2)}{(4x^2+5)}}\\ \amp= \lim_{x\to-\infty}\frac{-3+2/x}{\sqrt{4+5/x^2}}=\frac{-3}{\sqrt{4}}=\frac{-3}{2}. \end{align*}
