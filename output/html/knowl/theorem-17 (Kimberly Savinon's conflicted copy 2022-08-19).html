Theorem 3.132. Derivatives of Inverse Trigonometric Functions.

\begin{equation} \frac{d}{dx} \sin^{-1}x=\frac{1}{\sqrt{1−x^2}}\tag{3.7.1} \end{equation}
\begin{equation} \frac{d}{dx} \cos^{-1}x=\frac{−1}{\sqrt{1−x^2}}\tag{3.7.2} \end{equation}
\begin{equation} \frac{d}{dx} \tan^{-1}x=\frac{1}{1+x^2}\tag{3.7.3} \end{equation}
\begin{equation} \frac{d}{dx}\cot^{-1}x=\frac{−1}{1+x^2}\tag{3.7.4} \end{equation}
\begin{equation} \frac{d}{dx} \sec^{-1}x=\frac{1}{|x|\sqrt{x^2−1}}\tag{3.7.5} \end{equation}
\begin{equation} \frac{d}{dx} \csc^{-1}x=\frac{−1}{|x|\sqrt{x^2−1}}\tag{3.7.6} \end{equation}
